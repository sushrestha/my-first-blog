The algorithm for encryption was simple shift-ceaser algorithm. The letter is moved to next characters on the basis of position. Below is the algorithm used: 

        function doEncrypt(p){
        var input = p;
        encrypt ="";
        for(i=0;i<input.length;i++){
            asc = input.charCodeAt(i);
            asc += i*2;
            char_acs = String.fromCharCode(asc);
            encrypt += char_acs;
            }
        return encrypt;
    }
